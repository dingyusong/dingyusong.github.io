236. 二叉树的最近公共祖先-127#

难度中等685

给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。

百度百科中最近公共祖先的定义为：“对于有根树 T 的两个结点 p、q，最近公共祖先表示为一个结点 x，满足 x 是 p、q 的祖先且 x 的深度尽可能大（一个节点也可以是它自己的祖先）。”

例如，给定如下二叉树: root = [3,5,1,6,2,0,8,null,null,7,4]

示例 1:

输入: root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 1
输出: 3
解释: 节点 5 和节点 1 的最近公共祖先是节点 3。

示例 2:

输入: root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 4
输出: 5
解释: 节点 5 和节点 4 的最近公共祖先是节点 5。因为根据定义最近公共祖先节点可以为节点本身。

说明:

  • 所有节点的值都是唯一的。
  • p、q 为不同节点且均存在于给定的二叉树中。

后续遍历

考虑通过递归对二叉树进行先序遍历，当遇到节点 p 或 q时返回。从底至顶回溯，当节点 p, qp,q 在节点 root 的异侧时，节点 root 即为最近公共祖先，则向上返回 root 。

:rtype: TreeNode

二叉树遍历的方法中，后续遍历是先遍历子节点，后遍历根节点，从最底层往上层搜索。正好能用来找最近的祖先。

if root == None or root == p or root == q:
return root
left = self.lowestCommonAncestor(root.left,p,q)

#2. right不为空，说明右子树找到p或q
#3. 左右都不为空说明同时找到了p和q
#那么这个root节点就是我们想要的返回回去！
if left and right:

和上一个比较，这个解法递归不回溯。记录递归的路径。

  1. 广度优先遍历 记录到两个节点的路径（使用一个hash记录parent信息）
  2. 逆序推出路径。（从节点一直到INT-MIN）
  3. 从后往前找是否在另一个路径hash中。
/**
* struct TreeNode {
* int val;
* struct TreeNode *left;
* struct TreeNode *right;
* };
*/
class Solution {
public:
/**
*
* @param root TreeNode类
* @param o1 int整型
* @param o2 int整型
* @return int整型
*/
int lowestCommonAncestor(TreeNode* root, int o1, int o2) {
// write code here
/*
1：广度优先遍历，找到到o1和o2的路径（这期间用hash记录自己的父亲的信息）
2. 将其中之一的路径放到set中，另一个逆序虚查找hash，查找是不是在set中，这样就能找到最近的祖先了
*/
unordered_map<int, int> parent;
parent[root->val] = INT_MIN;
queue<TreeNode *> que;
que.push(root);
while(!parent.count(o1) || !parent.count(o2)) {
TreeNode *x = que.front();
que.pop();
if(x->left){
parent[x->left->val] = x->val;
que.push(x->left);
}
if(x->right){
parent[x->right->val] = x->val;
que.push(x->right);
}
}
set<int> path1;
int p = o1;
while(p != INT_MIN) {
path1.insert(p);
p = parent[p];
}
p = o2;
while(p != INT_MIN) {
if(path1.count(p)){
return p;
}
p = parent[p];
}
return -1;
}
};

if (root == null || !isBalanced)
return 0;
int left = height(root.left);
int right = height(root.right);

/**
* Definition for a binary tree node.
* struct TreeNode {
* int val;
* TreeNode *left;
* TreeNode *right;
* TreeNode() : val(0), left(nullptr), right(nullptr) {}
* TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
* TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
* };
*/
class Solution {
private:
vector<vector<int>> ret;
public:
vector<vector<int>> levelOrder(TreeNode* root) {
//异常判断
vector<vector<int>> ret;
if(!root) {
return ret;
}
stack<TreeNode *> sk1,sk2; //sk1存放当前层的内容，sk2是过渡存储sk1到sk2.
sk1.push(root);
while(!sk1.empty()) {
vector<int> levelRet;
//拿出栈1的内容，并且将其出队push到栈2
while(!sk1.empty()) {
TreeNode *node = sk1.top();
sk1.pop();
levelRet.push_back(node->val);
sk2.push(node);
}
ret.push_back(levelRet);
while(!sk2.empty()) {
TreeNode *node = sk2.top();
sk2.pop();
if(node->left) sk1.push(node->left);
if(node->right) sk1.push(node->right);
}
}
return ret;
}
};

给定一个非空二叉树，返回其最大路径和。

本题中，路径被定义为一条从树中任意节点出发，达到任意节点的序列。该路径至少包含一个节点，且不一定经过根节点。

示例 1:

输入: [1,2,3]

/ \
2 3
输出: 6

示例 2:

输入: [-10,9,20,null,null,15,7]
-10
/ \
9 20
/ \
15 7
输出: 42

/**
树中节点数目范围是 [1, 3 * 104]
-1000 <= Node.val <= 1000
* Definition for a binary tree node.
* struct TreeNode {
* int val;
* TreeNode *left;
* TreeNode *right;
* TreeNode() : val(0), left(nullptr), right(nullptr) {}

* TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
* };
*/
class Solution {
/*
一棵树分6中情况
1. 根
2. 根+左
3. 根+右
1，2，3可以回溯结合父节点
4. 根+左+右
5. 左
6. 右
4，5，6不可回溯结合父节点
*/
private:
int ret;
int dfs(TreeNode* root) {
//递归结束
if(root == nullptr) return INT_MIN>>2;//注意这个值，题目中有限制条件：-1000 <= Node.val <= 1000
int left = dfs(root->left);
int right = dfs(root->right);
//回溯分治
int res = max(max(left,right),root->val+left+right);
ret = max(res, ret);
return max(root->val,max(root->val+left,root->val+right));
}
public:
int maxPathSum(TreeNode* root) {
ret = INT_MIN>>1;
int r1 = dfs(root);
return max(r1,ret);
}
};

import sys
class Solution:
result = -sys.maxsize-1
def maxPathSum(self, root):
"""
:type root: TreeNode
:rtype: int
"""
self.maxValue(root)
return self.result
"""
最大路径和：根据当前节点的角色，路径和可分为两种情况：
一：以当前节点为根节点
1.只有当前节点
2.当前节点+左子树
3.当前节点+右子书
4.当前节点+左右子树
这四种情况的最大值即为以当前节点为根的最大路径和
此最大值要和已经保存的最大值比较，得到整个树的最大路径值
二：当前节点作为父节点的一个子节点
和父节点连接的话则需取【单端的最大值】
1.只有当前节点
2.当前节点+左子树
3.当前节点+右子书
这三种情况的最大值
"""
def maxValue(self,root):
if root == None:
return 0
leftValue = self.maxValue(root.left)
rightValue = self.maxValue(root.right)
value1 = root.val
value2 = root.val + leftValue
value3 = root.val + rightValue
value4 = root.val + rightValue + leftValue
#以此节点为根节点的最大值
maxValue = max([value1,value2,value3,value4])
#当前遍历树的最大值
self.result = max(maxValue, self.result)
#要和父节点关联，则需要取去除情况4的最大值
return max([value1,value2,value3])

不用看官方题解，那么复杂。 所有树的题目，都想成一颗只有根、左节点、右节点 的小树。然后一颗颗小树构成整棵大树，所以只需要考虑这颗小树即可。接下来分情况， 按照题意：一颗三个节点的小树的结果只可能有如下6种情况：

  1. 根 + 左 + 右
  2. 根 + 左
  3. 根 + 右
  4. 根
  5. 左
  6. 右

好了，分析上述6种情况， 只有 2,3,4 可以向上累加，而1,5,6不可以累加（这个很好想，情况1向上累加的话，必然出现分叉，情况5和6直接就跟上面的树枝断开的，没法累加），所以我们找一个全局变量存储 1,5,6这三种不可累加的最大值， 另一方面咱们用遍历树的方法求2,3,4这三种可以累加的情况。 最后把两类情况得到的最大值再取一个最大值即可。

class Solution(object):
def maxPathSum(self, root):
"""
:type root: TreeNode
:rtype: int
"""
self.max_sum = -sys.maxsize - 1
def scan(root):
if root is None:
return -sys.maxsize - 1
left = scan(root.left)
right = scan(root.right)
self.max_sum = max(self.max_sum, root.val + left + right, left, right) # 情况1,5,6，不累加直接放变量里暂存
return max(root.val, root.val + left, root.val + right) # 情况2,3,4 ，累加需要递归
new_max = scan(root)
return max(self.max_sum, new_max) # 两类情况再求最大

/**
* Definition for a binary tree node.
* struct TreeNode {

* TreeNode *left;

* };

/*
中序遍历是左中右。左先入栈，往后退一个左，左考察完了就改考察右了。
*/
class Solution {
public:
vector<int> inorderTraversal(TreeNode* root) {
vector<int> ret;
if(root == nullptr) return ret;
stack<TreeNode *> sk;
TreeNode *curr = root;
while(!sk.empty() || curr != nullptr) {
while(curr) {
sk.push(curr);
curr = curr->left;
}
TreeNode *node = sk.top();
sk.pop();
ret.push_back(node->val);
curr = node->right;
}
return ret;
}
};

dfs(sequence, 0, sequence.size()-1);

}
void dfs(vector<int> &sequence, int l, int r) {
if(l >= r || ret == false) {
return;
}
int root = sequence[r];
int l_cur = l, r_cur = r-1;
while(true) {
while(l_cur < r && sequence[l_cur] < root) l_cur++;
while(r_cur >= l && sequence[r_cur] > root) r_cur--;
if(l_cur <= r_cur){
ret = false;
}
break;
}
dfs(sequence, l, l_cur-1);
dfs(sequence, r_cur+1, r-1);//切记一定要搞清楚边界！
}
};

105. 从前序与中序遍历序列构造二叉树#

/**
* Definition for binary tree
* struct TreeNode {
* int val;
* TreeNode *left;
* TreeNode *right;
* TreeNode(int x) : val(x), left(NULL), right(NULL) {}
* };
*/
class Solution {
private:
std::map<int, int> indexsForInOrder;
public:
TreeNode* reConstructBinaryTree(vector<int> pre,vector<int> vin) {
/*
前序遍历：根左右
中序遍历：左根右
每次需要确定的是左子树的边界，右子树的边界和根。（重复性）
*/
//保存in的索引值
for(int i = 0; i < vin.size(); i++) {
indexsForInOrder.insert({vin[i],i});
}
return reConstructBinaryTree(pre, 0,pre.size()-1,0);
}
TreeNode* reConstructBinaryTree(vector<int> pre, int preL, int preR,int inL) {
/*
前序遍历：根左右
中序遍历：左根右
每次需要确定的是左子树的边界，右子树的边界和根。（重复性）
*/
if(preL > preR) {
//如果他俩相等的话，则就是根节点！
return NULL;
}
//前序遍历是根左右！
TreeNode *root = new TreeNode(pre[preL]);
//拿到根节点在中序遍历中的索引
int rootIndex = indexsForInOrder.find(pre[preL])->second;
//inL 是为了确认每一课树的起点，前序遍历的起点
int leftTreeSize = rootIndex - inL;
//可以把前序遍历分为两段！左子树和右子树
root->left = reConstructBinaryTree(pre, preL+1,preL+leftTreeSize, inL);
root->right = reConstructBinaryTree(pre, preL+leftTreeSize+1,preR, inL + leftTreeSize + 1);
return root;
}
};

# Definition for a binary tree node.
# class TreeNode:
# def __init__(self, x):
# self.val = x
# self.left = None
# self.right = None
class Solution:
index = 0
def kthSmallest(self, root, k):
"""
:type root: TreeNode
:type k: int
:rtype: int
"""
res = 0
if root == None:
return None
res = self.kthSmallest(root.left,k)
if self.index == k:
return res
self.index += 1
if self.index == k:
return root.val
return self.kthSmallest(root.right,k)

# Definition for a binary tree node.
# class TreeNode:
# def __init__(self, x):
# self.val = x
# self.left = None
# self.right = None
class Solution:
#递归深度优先搜索。出口在节点，节点的left或者right = 0--以节点为视角。
def maxDepth(self, root):
"""
:type root: TreeNode
:rtype: int
"""
if root is None:
return 0
else:
left_height = self.maxDepth(root.left)
right_height = self.maxDepth(root.right)
return max(left_height, right_height) + 1;

235. 二叉搜索树的最近公共祖先#

难度简单350

给定一个二叉搜索树, 找到该树中两个指定节点的最近公共祖先。

百度百科中最近公共祖先的定义为：“对于有根树 T 的两个结点 p、q，最近公共祖先表示为一个结点 x，满足 x 是 p、q 的祖先且 x 的深度尽可能大（一个节点也可以是它自己的祖先）。”

例如，给定如下二叉搜索树: root = [6,2,8,0,4,7,9,null,null,3,5]

示例 1:

输入: root = [6,2,8,0,4,7,9,null,null,3,5], p = 2, q = 8
输出: 6
解释: 节点 2 和节点 8 的最近公共祖先是 6。

示例 2:

输入: root = [6,2,8,0,4,7,9,null,null,3,5], p = 2, q = 4
输出: 2
解释: 节点 2 和节点 4 的最近公共祖先是 2, 因为根据定义最近公共祖先节点可以为节点本身。

说明:

  • 所有节点的值都是唯一的。
  • p、q 为不同节点且均存在于给定的二叉搜索树中。
# Definition for a binary tree node.
# class TreeNode:
# def __init__(self, x):
# self.val = x
# self.left = None
# self.right = None
class Solution:
def lowestCommonAncestor(self, root, p, q):
"""
:type root: TreeNode
:type p: TreeNode
:type q: TreeNode
:rtype: TreeNode
"""
#递归出口
if root == None or root == p or root == q:
return root
left = self.lowestCommonAncestor(root.left,p,q)
right = self.lowestCommonAncestor(root.right,p,q)
if left and right:
return root
return left or right

/**
* Definition for a binary tree node.
* struct TreeNode {
* int val;
* TreeNode *left;
* TreeNode *right;
* TreeNode() : val(0), left(nullptr), right(nullptr) {}
* TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
* TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
* };
*/
class Solution {
private:
bool dfs(TreeNode* left, TreeNode* right) {
if(left == nullptr && right == nullptr) {
return true;
}
if(left == nullptr || right == nullptr) {
return false;
}
if(left->val != right->val) {
return false;
}
return dfs(left->left,right->right) & dfs(left->right, right->left);
}
public:
bool isSymmetric(TreeNode* root) {
if(root == nullptr) {
return false;
}
return dfs(root->left,root->right);
}
};

理解#

递归回溯#

参考：236. 二叉树的最近公共祖先-127

  1. return 在递归调用之前是纯递归，return在递归调用之后是回溯，回来之后拿到相应的结果。分治必然依赖回溯！
  2. 纯递归的感觉是下一层楼消失了，他把自己的生产给了外界（全局变量，引用参数），或者就没有结果（这种不太会有，因为遍历总是有目的的）。
  3. 回溯是下一层楼消失之后，他通过return把结果给了上层，上层来处理数据，然后本层消失的时候，继续把结果往上抛（return），所有的楼层都消失之后（函数递归栈都弹出了）就返回了本地递归调用的结果。
  4. 回溯的递归函数有返回值，纯递归的函数没有返回值（void）。纯递归将结果给了外界，或者是引用参数。
  5. 已知张三和李四都在中国！请问他们的最小行政范围。从上往下查，找到之后从下往上报，然后层层汇总，比如张三在信阳，李四在郑州，则汇总到河南的时候，那河南必然是他们倆的最小行政单位。
  6. 分治-回溯。某些情况无法回溯，那么就需要借助栈外的变量。例如：124：二叉树的最大路径和！（打开思路，有全局保存的，也有回溯的，最后综合考虑。）
