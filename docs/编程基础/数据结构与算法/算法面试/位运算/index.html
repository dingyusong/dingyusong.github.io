位运算

剑指 Offer 15. 二进制中1的个数 - 3#

位运算：

  1. 右移 >>
  2. 与 |

(x >> n)&1取第n位的值0或者1。

class Solution {
public:
int NumberOf1(int n) {
int count = 0;
for(int i = 0; i<32; i++) {
if((n >> i)&1){
count++;
}
}
return count;
}
};

通用解法，不用管它有多少位。可以适配不同位数系统。

class Solution {
public:
int NumberOf1(int n) {
int count = 0;
while(n != 0) {
if((n >> 1)&1) count++;
n >>= 1;
}
return count;
}
};

位运算：n & (n-1) 删除最右边的1。

class Solution {
public:
int NumberOf1(int n) {
int count = 0;
while(n != 0) {
count++;
n = n & (n-1);
}
return count;
}
};

136. 只出现一次的数字 - 39#

class Solution {
public:
/**
* 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
*
*
* @param array int整型vector
* @return int整型vector
*/
vector<int> FindNumsAppearOnce(vector<int>& array) {
vector<int> ret;
// unordered_set<int> s;
set<int> s;
for(int v: array) {
if(s.count(v)){
s.erase(v);
}else{
s.insert(v);
}
}
for(int num: s) {
ret.push_back(num);
}
return ret;
}
};
class Solution {
public:
/**
* 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
*
*
* @param array int整型vector
* @return int整型vector
*/
vector<int> FindNumsAppearOnce(vector<int>& array) {
//得到两个不同的数的异或
int tmp = 0;
for(int num:array) {
tmp ^= num;
}
/*
从低位到高位，找到第一个1，也就是两个数不同的地方
（可以根据这个不同的地方，将数组分为两组，那两个不同的数，必然分布在，这两个组中）
然后进行分组异或即可得到这两个数
*/
int mask = 1;
while((tmp&mask) == 0) {//注意：位运算一定要加括号。将他们括起来。
mask <<= 1;
}
int a = 0,b = 0;
for(int num : array) {
if((num&mask) == 0) {
a ^= num;
}
else {
b ^= num;
}
}
return a < b ? vector<int>({a,b}):vector<int>({b,a});
}
};

剑指 Offer 65. 不用加减乘除做加法 - 7#

class Solution {
public:
int Add(int num1, int num2) {
//进位是取&右移一位
//无进位和就是异或。
//他们俩在相加，直到进位为0，结束！
while(num2 != 0) {
int c = (num1&num2) << 1;
num1 ^= num2;//如果没有进位，异或的结果就是加法的结果
num2 = c;//如果进位为0，异或的结果就是加法的结果
}
return num1;
}
};
