/**
* Definition for singly-linked list.
* struct ListNode {
* int val;
* ListNode *next;
* ListNode() : val(0), next(nullptr) {}
* ListNode(int x) : val(x), next(nullptr) {}
* ListNode(int x, ListNode *next) : val(x), next(next) {}
* };
*/
class Solution {
public:
ListNode* reverseList(ListNode* head) {
ListNode *pre = nullptr;
ListNode *curr = head; //当前考察的节点
while(curr) {
ListNode *next = curr->next;
curr->next = pre;
pre = curr;
curr = next;
}
return pre;
}
};

25. K 个一组翻转链表#

/**
* Definition for singly-linked list.
* struct ListNode {
* int val;
* ListNode *next;
* ListNode() : val(0), next(nullptr) {}
* ListNode(int x) : val(x), next(nullptr) {}
* ListNode(int x, ListNode *next) : val(x), next(next) {}
* };
*/
class Solution {
private:
ListNode *reverse(ListNode *head) {
ListNode *pre = nullptr; //前驱指针
ListNode *curr = head; //当前要反转的节点
while(curr) {
ListNode *next = curr->next;
curr->next = pre;
pre = curr;
curr = next;
}
return pre;
}
public:
ListNode* reverseKGroup(ListNode* head, int k) {
if(head == nullptr) {
return nullptr;
}
ListNode *dummyHead = new ListNode(-1,head);
ListNode *pre = dummyHead;//指向待反转链表前驱。
ListNode *end = dummyHead;//指向待反转链表结尾。
while(end->next) {//说明此时end是最后一个，没有必要反转了
int i = 0;
while(i < k && end) {
end = end->next;
i++;
}
if(!end) {
return dummyHead->next;
}
ListNode *next = end->next; //保留下一个待考察区域的头指针
end->next = nullptr;
ListNode *start = pre->next;
pre->next = reverse(start); //返回的尾部那个指针接到前面
start->next = next;
//下一轮的处理
pre = start;//指向待反转链表前驱。
end = start;//指向待反转链表前驱。下轮循环让其指向待翻转的结尾
}
return dummyHead->next;
}
};

两个指针同时往后走，先把短的走完。长的也走了端的长度。然后在长的上走（长的-短的）距离，长的走完。第二次遍历的长的剩下的长度和短的一样。

/**

* };

class Solution {

if(headA == headB) {
return headB;
}
ListNode *p1 = headA, *p2 = headB;
while(p1 && p2) {
if(p1 == p2) {
return p1;
}
p1 = p1->next;
p2 = p2->next;
}
if(p1){
p2 = headA;
while(p1){
p1 = p1->next;
p2 = p2->next;
}
p1 = headB;
}else if(p2) {
p1 = headB;
while(p2){
p1 = p1->next;
p2 = p2->next;
}
p2 = headA;
}
while(p1 != p2) {
p1 = p1->next;
p2 = p2->next;
}
return p1;
}
};

return headB;
}
ListNode *pA = headA, *pB = headB;
while(pA != pB) {
pA = pA == nullptr? headB:pA->next;
pB = pB == nullptr? headA:pB->next;
}
return pA;
}

您可以假设除了数字 0 之外，这两个数都不会以 0 开头。

输出：7 -> 0 -> 8
原因：342 + 465 = 807
* ListNode next;
* ListNode(int x) { val = x; }
* }
*/
class Solution {
public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
ListNode pre = new ListNode(0);
ListNode cur = pre;
int carry = 0;
while(l1 != null || l2 != null) {
int x = l1 == null ? 0 : l1.val;
int y = l2 == null ? 0 : l2.val;
int sum = x + y + carry;
carry = sum / 10;
sum = sum % 10;
cur.next = new ListNode(sum);
cur = cur.next;
if(l1 != null)
l1 = l1.next;
if(l2 != null)
l2 = l2.next;
}
if(carry == 1) {
cur.next = new ListNode(carry);
}
return pre.next;
}
}

/** * Definition for singly-linked list. * struct ListNode { * int val; * ListNode *next; * ListNode() : val(0), next(nullptr) {} * ListNode(int x) : val(x), next(nullptr) {} * ListNode(int x, ListNode *next) : val(x), next(next) {} * }; */class Solution {private: ListNode *mergeList(ListNode* l, ListNode* r) { ListNode *dummyHead = new ListNode(-1); ListNode *p = dummyHead;//p指向最后一个排好的链表节点 while(l && r) { if(l->val < r->val) { p->next = l; l = l->next; }else { p->next = r; r = r->next; } p = p->next; } if(l) p->next = l; if(r) p->next = r; return dummyHead->next; }public: //再次理解递归分治! 字底向上归并。

  1. 循环一遍获得链表的长度
  2. 迭代(合并的两个节点从len=1开始，每次翻倍，直到大于链表的长度)
  3. 在上面的每一次的迭代中，用一个curr指针指向待处理的节点。取两个sublength长度，进行合并。
  4. 合并两个有序链表是基本操作了。
class Solution {
// 自底向上归并排序
public ListNode sortList(ListNode head) {
if(head == null){
return head;
}
// 1. 首先从头向后遍历,统计链表长度
int length = 0; // 用于统计链表长度
ListNode node = head;
while(node != null){
length++;
node = node.next;
}
// 2. 初始化 引入dummynode
ListNode dummyHead = new ListNode(0);
dummyHead.next = head;
// 3. 每次将链表拆分成若干个长度为subLen的子链表 , 并按照每两个子链表一组进行合并
for(int subLen = 1;subLen < length;subLen <<= 1){ // subLen每次左移一位（即sublen = sublen*2） PS:位运算对CPU来说效率更高
ListNode prev = dummyHead;
ListNode curr = dummyHead.next; // curr用于记录拆分链表的位置
while(curr != null){ // 如果链表没有被拆完
// 3.1 拆分subLen长度的链表1
ListNode head_1 = curr; // 第一个链表的头 即 curr初始的位置
for(int i = 1; i < subLen && curr != null && curr.next != null; i++){ // 拆分出长度为subLen的链表1
curr = curr.next;
}
// 3.2 拆分subLen长度的链表2
ListNode head_2 = curr.next; // 第二个链表的头 即 链表1尾部的下一个位置
curr.next = null; // 断开第一个链表和第二个链表的链接
curr = head_2; // 第二个链表头 重新赋值给curr
for(int i = 1;i < subLen && curr != null && curr.next != null;i++){ // 再拆分出长度为subLen的链表2
curr = curr.next;
}
// 3.3 再次断开 第二个链表最后的next的链接
ListNode next = null;
if(curr != null){
next = curr.next; // next用于记录 拆分完两个链表的结束位置
curr.next = null; // 断开连接
}
// 3.4 合并两个subLen长度的有序链表
ListNode merged = mergeTwoLists(head_1,head_2);
prev.next = merged; // prev.next 指向排好序链表的头
while(prev.next != null){ // while循环 将prev移动到 subLen*2 的位置后去
prev = prev.next;
}
curr = next; // next用于记录 拆分完两个链表的结束位置
}
}
// 返回新排好序的链表
return dummyHead.next;
}
// 此处是Leetcode21 --> 合并两个有序链表
public ListNode mergeTwoLists(ListNode l1,ListNode l2){
ListNode dummy = new ListNode(0);
ListNode curr = dummy;
while(l1 != null && l2!= null){ // 退出循环的条件是走完了其中一个链表
// 判断l1 和 l2大小
if (l1.val < l2.val){
// l1 小 ， curr指向l1
curr.next = l1;
l1 = l1.next; // l1 向后走一位
}else{
// l2 小 ， curr指向l2
curr.next = l2;
l2 = l2.next; // l2向后走一位
}
curr = curr.next; // curr后移一位
}
// 退出while循环之后,比较哪个链表剩下长度更长,直接拼接在排序链表末尾
if(l1 == null) curr.next = l2;
if(l2 == null) curr.next = l1;
// 最后返回合并后有序的链表
return dummy.next;
}
}

return NULL;
}
ListNode *cur1 = pHead1;

//递归是深度优先遍历，注意递归的结束条件即本层的逻辑处理。
class Solution {
public:
ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
//递归结束条件，递归边界，满足条件的边界结果
if(list1 == nullptr) {
return list2;
}
if(list2 == nullptr) {
return list1;
}
//当前层逻辑处理
if(list1->val < list2->val) {
//下探后回来
list1->next = mergeTwoLists(list1->next, list2);
return list1;
}
else {
//下探后回来
list2->next = mergeTwoLists(list2->next, list1);
return list2;
}
}
};

# self.next = None
class Solution:
def rotateRight(self, head, k):
"""
:type head: ListNode
:type k: int
:rtype: ListNode
"""
#1.循环或得链表长度和尾指针
p,tail, n = head ,head, 0
while p:
n += 1
tail = p
p = p.next
if n == 0 or n == 1:
return head
shift = k%n
if shift == 0:
return head
#2.截断链表，然后拼接
shift = n - shift
p , q= head, head
for i in range(shift):
q = p
p = p.next
q.next = None
tail.next = head
return p
  1. 循环得到链表的长度和尾指针。
  2. 截断链表，然后拼接。

141. 环形链表#

给定一个链表，判断链表中是否有环。

为了表示给定链表中的环，我们使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。 如果 pos 是 -1，则在该链表中没有环。

示例 1：

输入：head = [3,2,0,-4], pos = 1
输出：true
解释：链表中有一个环，其尾部连接到第二个节点。

示例 2：

输入：head = [1,2], pos = 0
输出：true
解释：链表中有一个环，其尾部连接到第一个节点。

示例 3：

输入：head = [1], pos = -1
输出：false
解释：链表中没有环。

进阶：

你能用 O(1)（即，常量）内存解决此问题吗？

# Definition for singly-linked list.
# class ListNode(object):
# def __init__(self, x):
# self.val = x
# self.next = None
class Solution(object):
def hasCycle(self, head):
"""
:type head: ListNode
:rtype: bool
"""
if head == None or head.next ==None:
return False
slow = head
fast = head.next
#如果是环形赛道，两个速度不同的人必定相遇，则可以出循环
while slow != fast:
#如果非环境赛道，则有终点且，快的人先走到终点
if fast == None or fast.next == None:
return False
fast = fast.next.next
slow = slow.next
return True

快慢指针：

  1. 如果有环必定相遇。
  2. 如果无环，快指针先到终点。

142. 环形链表 II#

难度中等569

给定一个链表，返回链表开始入环的第一个节点。 如果链表无环，则返回 null。

为了表示给定链表中的环，我们使用整数 pos 来表示链表尾连接到链表中的位置（索引从 0 开始）。 如果 pos 是 -1，则在该链表中没有环。

说明：不允许修改给定的链表。

示例 1：

输入：head = [3,2,0,-4], pos = 1
输出：tail connects to node index 1
解释：链表中有一个环，其尾部连接到第二个节点。

示例 2：

输入：head = [1,2], pos = 0
输出：tail connects to node index 0
解释：链表中有一个环，其尾部连接到第一个节点。

示例 3：

输入：head = [1], pos = -1
输出：no cycle
解释：链表中没有环。

进阶： 你是否可以不用额外空间解决此题？

# Definition for singly-linked list.
# class ListNode(object):
# def __init__(self, x):
# self.val = x
# self.next = None
class Solution(object):
def detectCycle(self, head):
"""
:type head: ListNode
:rtype: ListNode
"""
i = 0
listSet = set()
while head:
if head in listSet:
return head
else:
listSet.add(head)
head = head.next
return None

思路：快慢指针。

  1. 相遇的时候快比慢多走n圈。
  2. 快的长度是慢的2倍。

第一次相遇的时候。满指针走过的路程是1圈的长度。第二次相遇，同速度走则是入口的位置。因为第一次走的长度是圈外长度+入圈长度。这个等于一个圈的长度。所以一个圈的长度减去入圈长度，也就是剩下的圈的长度，等于圈外长度。
