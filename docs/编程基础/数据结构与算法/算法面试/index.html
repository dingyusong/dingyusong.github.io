class Solution {
public:
int mySqrt(int x) {
if(x==0 || x == 1) {
return x;
}
int l = 1, r = x/2;
while(l < r) {
int mid = l + (r-l + 1)/2;
/*
明确二分的判断，如果
一个数的平方大于x，那他一定不是x的平方根
一个数的平方小于等于x，那他可能是x的平方根
根据二分的定义，确定下一次的边界。
*/
if(mid > x/mid) {
r = mid - 1;//舍弃mid，mid不可能是解
}
else if(mid < x/mid) {
l = mid;//保留mid，mid可能是解
}else {
return mid;
}
}
return l;
}
};

class Solution:
def threeSum(self, nums):
"""
:type nums: List[int]
:rtype: List[List[int]]
"""
if len(nums) < 3:
return []
nums_hash = {}
result = list()
#以数组值为key，出现次数为value，创建一个hash表
for num in nums:
nums_hash[num] = nums_hash.get(num, 0) + 1
#如果hash表中存在0，且0出现的次数大于3，则有一个解为[0,0,0]
if 0 in nums_hash and nums_hash[0] >= 3:
result.append([0, 0, 0])
#获得升序排序数组
nums = sorted(list(nums_hash.keys()))
#分成负数和正数两组
pos , neg = [], []
for i in nums:
if i > 0:
pos.append(i)
elif i < 0:
neg.append(i)
for i in neg:
for j in pos:
#分离出相反数
if (i + j == 0) and (0 in nums_hash) and (nums_hash[0] >= 1):
result.append([i, 0, j])
continue
#寻找差值
dif = 0 - i- j
#前提条件是dif在hash表中
if dif in nums_hash:
#差值在两个之中，则就要判断 差值是否有两个以上
if dif in (i,j):
if nums_hash[dif] >= 2:
result.append([i,dif,j])
else:
#去重
if (dif < 0 and dif > i) or (dif > 0 and dif < j):
result.append([i,dif,j])
return result

if(s.size() == 0) return 0;
int n = s.size();
int maxlen = 0;
unordered_set<char> charSet;
int start = 0;
for(int i = 0; i < n; i++) {
char ch = s[i];
/*
注意：窗口往前滑动，切记一定是while！！！

charSet.erase(s[start]);
start++;
}
//滑动的过程中做记录
maxlen = max(maxlen, i-start+1);
charSet.insert(ch);
}
return maxlen;
}
};

提示：题目数据保证数组之中任意元素的全部前缀元素和后缀（甚至是整个数组）的乘积都在 32 位整数范围内。

说明: 请不要使用除法，且在 O(n) 时间复杂度内完成此题。

进阶： 你可以在常数空间复杂度内完成这个题目吗？（ 出于对空间复杂度分析的目的，输出数组不被视为额外空间。）

class Solution {
public int[] productExceptSelf(int[] nums) {
int[] res = new int[nums.length];
int k = 1;
for(int i = 0; i < res.length; i++){
res[i] = k;
k = k * nums[i]; // 此时数组存储的是除去当前元素左边的元素乘积
}
k = 1;
for(int i = res.length - 1; i >= 0; i--){
res[i] *= k; // k为该数右边的乘积。

}

}
