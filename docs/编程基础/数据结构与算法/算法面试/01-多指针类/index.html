3. 无重复字符的最长子串-277#

给定一个字符串 s ，请你找出其中不含有重复字符的 最长子串 的长度。

思路：字符串是字符数组。按照数组的思路来解决。

class Solution {
public:
int lengthOfLongestSubstring(string s) {
if(s.size() == 0) return 0;
int n = s.size();
int maxlen = 0;
unordered_set<char> charSet;
int start = 0;
for(int i = 0; i < n; i++) {
char ch = s[i];
/*
注意：窗口往前滑动，切记一定是while！！！
如果set里面能找到ch就把左边的删了，知道删除ch！
*/
while(charSet.find(ch) != charSet.end()) {
charSet.erase(s[start]);
start++;
}
//滑动的过程中做记录
maxlen = max(maxlen, i-start+1);
charSet.insert(ch);
}
return maxlen;
}
};

76. 最小覆盖子串 - 65#

给你一个字符串 s 、一个字符串 t 。返回 s 中涵盖 t 所有字符的最小子串。如果 s 中不存在涵盖 t 所有字符的子串，则返回空字符串 "" 。

注意：

对于 t 中重复字符，我们寻找的子字符串中该字符数量必须不少于 t 中该字符数量。 如果 s 中存在这样的子串，我们保证它是唯一的答案。

示例 1：

输入：s = "ADOBECODEBANC", t = "ABC"
输出："BANC"

示例 2：

输入：s = "a", t = "a"
输出："a"

示例 3:

输入: s = "a", t = "aa"
输出: ""
解释: t 中两个字符 'a' 均应包含在 s 的子串中，
因此没有符合条件的子字符串，返回空字符串。

来源：力扣（LeetCode） 链接：https://leetcode-cn.com/problems/minimum-window-substring 著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。

class Solution {
public:
string minWindow(string s, string t) {
int l = 0, r = 0, start = 0, size = INT_MAX;
int count = t.size();//t的字符数量
vector<int>need(128,0);
for(char c: t) {
need[c]++; //need说明我们在窗口中需要他
}
while(r < s.size()) {
/*
考察r：
r加入之后会出现几种情况：
1. 所有的元素都在窗口内
尝试减少窗口，获得并记录最小子串。
2. 没有达到目标 继续考察 r++
*/
char ch = s[r];
if(need[ch] > 0) count--;//找到了一个需要的
need[ch]--;
if(count == 0) {//目前窗口中包含所有的字符串
//尝试缩小字符串
while(need[s[l]] < 0 && l < r) {
need[s[l++]]++; //l往前移动
}
//此时窗口符合需求
if(size > r-l+1) { //更新答案
size = r-l+1;
start = l;
}
//继续试图寻找更小的窗口（左边界往前做一步）
need[s[l]]++;
l++;
count++; //待满足的count需要增加
}
r++;//右边界往前走一步
}
return size==INT_MAX?"":s.substr(start,size);
}
};

239. 滑动窗口最大值 - 61#

维护单调队列。

class Solution {
public:
vector<int> maxSlidingWindow(vector<int>& nums, int k) {
vector<int>ret(nums.size()-k+1);
deque<int> dq;
for(int i = 0; i < nums.size(); i++) {
while(!dq.empty() && nums[i] >= nums[dq.back()]) dq.pop_back();
dq.push_back(i);
//判断队首是否过期
if(dq.front() <= i-k) {
dq.pop_front();
}
//判断窗口是否形成
if(i+1 >= k) {
ret[i-k+1] = nums[dq.front()];
}
}
return ret;
}
};

给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有和为 0 且不重复的三元组。

示例 2：

示例 3：

输出：[]

class Solution {
public:
vector<vector<int>> threeSum(vector<int>& nums) {
vector<vector<int>> ret;
//0.边界处理
int n = nums.size();
if(n < 3) return ret;
//1.数组排序 注意数组排序！
sort(nums.begin(),nums.end());
if(nums[0] > 0) return ret;
//2.遍历数组，左右指针调整
int L,R;
for(int i = 0; i < n; i++) {
//注意i > 0在前面！否则会引起索引越界！：去除第一位的重复解
if(i > 0 && nums[i] == nums[i-1]) continue;
L = i+1;
R = n-1;
while(L < R) {
if(nums[i] + nums[L] + nums[R] == 0){
ret.push_back({nums[i],nums[L],nums[R]});
//找到一个结果之后，我们把可能重复的都要去掉！注意！
while(L<R && nums[L] == nums[L+1]) L++;//去除左边的重复
while(L<R && nums[R] == nums[R-1]) R--;//去除右边的重复
L++;
R--;
}
else if(nums[i] + nums[L] + nums[R] < 0) L++;//注意是：L++还是R--
else R--;
}
}
return ret;
}
};

class Solution:
def threeSum(self, nums):
"""
:type nums: List[int]
:rtype: List[List[int]]
"""
if len(nums) < 3:
return []
nums_hash = {}
result = list()
#以数组值为key，出现次数为value，创建一个hash表
for num in nums:
nums_hash[num] = nums_hash.get(num, 0) + 1
#如果hash表中存在0，且0出现的次数大于3，则有一个解为[0,0,0]
if 0 in nums_hash and nums_hash[0] >= 3:
result.append([0, 0, 0])
#获得升序排序数组
nums = sorted(list(nums_hash.keys()))
#分成负数和正数两组
pos , neg = [], []
for i in nums:
if i > 0:
pos.append(i)
elif i < 0:
neg.append(i)
for i in neg:
for j in pos:
#分离出相反数
if (i + j == 0) and (0 in nums_hash) and (nums_hash[0] >= 1):
result.append([i, 0, j])
continue
#寻找差值
dif = 0 - i- j
#前提条件是dif在hash表中
if dif in nums_hash:
#差值在两个之中，则就要判断 差值是否有两个以上
if dif in (i,j):
if nums_hash[dif] >= 2:
result.append([i,dif,j])
else:
#去重
if (dif < 0 and dif > i) or (dif > 0 and dif < j):
result.append([i,dif,j])
return result

class Solution {
public:
/**
* 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
*
*
* @param array int整型vector
* @return int整型vector
*/
vector<int> reOrderArray(vector<int>& array) {
/*
奇数 0 偶数
[l...j) 奇数
(o...r] 偶数
*/
int l = 0, j = l, r = array.size()-1, o = r;
while(j < o) {
while(j <= r && array[j]%2) j++;
while(o >= 0 && array[o]%2 == 0) o--;
if(j >= o){
break;
}
swap(array[o], array[j]);
}
return array;
}
};

void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
int i = nums1.size() - 1;//i指向待填入的那个位置
m--;//m,n指向那两个数组待处理部分的最后一个。
n--;
while (n >= 0) {
while (m >= 0 && nums1[m] > nums2[n]) {
swap(nums1[i--], nums1[m--]);
}
swap(nums1[i--], nums2[n--]);
}
}

26. 删除有序数组中的重复项#

两个指针。前面一个指针找到不重复的，后面的指针加1，然后后面的指针的数据覆盖前面的指针即可。

class Solution:
def removeDuplicates(self, nums):
"""
:type nums: List[int]
:rtype: int
"""
n = len(nums)
if n == 0:
return 0
i, j = 1, 0
while i < n:
if nums[i] != nums[j]:
j += 1;
nums[j] = nums[i]
i += 1
return j+1

75. 颜色分类#

3路快排的其中一步。讲一个数组分为3部分。

class Solution {
public:
void sortColors(vector<int>& nums) {
int l = 0, r = nums.size()-1;
/*
区间定义：
红色：[l lt]
白色：[lt+1, i)
蓝色：[gt, r]
*/
int lt = l-1, i = l, gt = r+1;
while(i < gt) {
if(nums[i] == 0) {
swap(nums[lt+1],nums[i]);
lt++;
i++;
}
else if(nums[i] == 1) {
i++;
}
else {
swap(nums[i], nums[gt-1]);
gt--;
}
}
}
};

912. 排序数组#

典型的分治思想

class Solution {
private:
void m_quick(vector<int>& nums, int l, int r){
/*
重点3：递归结束条件,是>=不要落下等于
*/
if(l >= r) {
return;
}
swap(nums[l],nums[rand()%(r-l+1) + l]);
int povit = nums[l];
/*
重点1：区间定义：[l+1 lt] < povit
[lt+1 i) = povit
[gt r] > povit
*/
int lt = l, i = l+1, gt = r+1;
while(i < gt) {
//迭代：对i进行考察
if(nums[i] < povit) {
swap(nums[i],nums[lt+1]);
lt++;
i++;
}else if(nums[i] == povit) {
i++;
}
else {
swap(nums[i],nums[gt-1]);
gt--;
}
}
swap(nums[l],nums[lt]);
//对左右两部分无序的进行排序
m_quick(nums,l,lt);
m_quick(nums,gt,r);
}
public:
vector<int> sortArray(vector<int>& nums) {
//注意：srand(time(NULL));
//重点2，随机标杆
srand(time(NULL));
m_quick(nums, 0, nums.size()-1);
return nums;
}
};

class Solution:
def threeSumClosest(self, nums, target):
"""
:type nums: List[int]
:type target: int
:rtype: int
"""
n = len(nums)
if n < 3:
return 0
#获得升序排序数组
nums = sorted(nums)
closeNum = nums[0] + nums[1] + nums[2]
for i in range(n-2):
l = i + 1
r = n -1
while l < r:
threeSum = nums[i] + nums[l] + nums[r]
if abs(threeSum - target) < abs(closeNum - target):
closeNum = threeSum
if threeSum > target:
r -= 1
elif threeSum < target:
l += 1
else:
return target
return closeNum

11. 盛最多水的容器#

class Solution:
def maxArea(self, height):
"""
:type height: List[int]
:rtype: int
"""
n = len(height)
maxArea = 0
#从两个顶点想内移动。
i, j = 0, n-1
while i < j:
if height[i] < height[j]:
area = (j - i)*height[i]
i += 1
else:
area = (j - i)*height[j]
j -= 1
maxArea = max(area, maxArea)
return maxArea

344. 反转字符串#

双指针从前往后、从后往前依次交换即可。

class Solution {
public void reverseString(char[] s) {
if(s == null || s.length == 0) return;
int left = 0;
int right = s.length-1;
while(left < right){ //奇数个的时候中间元素不动
char temp = s[left];
s[left] = s[right];
s[right] = temp;
left ++;
right --;
}
}
}
