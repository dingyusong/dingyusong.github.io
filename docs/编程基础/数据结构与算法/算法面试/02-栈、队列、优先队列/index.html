20. 有效的括号#

class Solution:
def isValid(self, s):
"""
:type s: str
:rtype: bool
"""
n = len(s)
listStack = []
for i in range(n):
if len(listStack) == 0:
listStack.append(s[i])
elif s[i] == ")" and listStack[-1] == "(":
listStack.pop()
elif s[i] == "}" and listStack[-1] == "{":
listStack.pop()
elif s[i] == "]" and listStack[-1] == "[":
listStack.pop()
else:
listStack.append(s[i])
if len(listStack) == 0:
return True
else:
return False

class MyQueue {
private:
stack<int> sk_in;
stack<int> sk_out;
public:
MyQueue() {
}
void push(int x) {
sk_in.push(x);
}
int pop() {
int ret = peek();
if(!sk_out.empty()) {
sk_out.pop();//细心一点别丢了分号
}
return ret;
}
int peek() {
if(sk_out.empty()) {
while(!sk_in.empty()) {
int val = sk_in.top();
sk_in.pop();
sk_out.push(val);
}
}
if(sk_out.empty()) return -1;
int ret = sk_out.top();
return ret;
}
bool empty() {
return sk_out.empty() && sk_in.empty();
}
};

模拟栈操作

/**
* 栈是一种数据结构，特点：先进后出
* 练习：使用全局变量模拟栈的操作
*/
#include <stdio.h>
#include <stdbool.h>
#include <assert.h>
//保护全局变量：在全局变量前加static后，这个全局变量就只能在本文件中使用
static int data[1024];//栈最多能保存1024个数据
static int count = 0;//目前已经放了多少个数(相当于栈顶位置)
//数据入栈 push
void push(int x){
assert(!full());//防止数组越界
data[count++] = x;
}
//数据出栈 pop
int pop(){
assert(!empty());
return data[--count];
}
//查看栈顶元素 top
int top(){
assert(!empty());
return data[count-1];
}
//查询栈满 full
bool full() {
if(count >= 1024) {
return 1;
}
return 0;
}
//查询栈空 empty
bool empty() {
if(count <= 0) {
return 1;
}
return 0;
}
int main(){
//入栈
for (int i = 1; i <= 10; i++) {
push(i);
}
//出栈
while(!empty()){
printf("%d ", top()); //栈顶元素
pop(); //出栈
}
printf("\n");
return 0;
}

155. 最小栈 - 56#

难度简单634收藏分享切换为英文关注反馈

设计一个支持 push ，pop ，top 操作，并能在常数时间内检索到最小元素的栈。

  • push(x) —— 将元素 x 推入栈中。
  • pop() —— 删除栈顶的元素。
  • top() —— 获取栈顶元素。
  • getMin() —— 检索栈中的最小元素。

示例:

输入：
["MinStack","push","push","push","getMin","pop","top","getMin"]
[[],[-2],[0],[-3],[],[],[],[]]
输出：
[null,null,null,null,-3,null,0,-2]
解释：
MinStack minStack = new MinStack();
minStack.push(-2);
minStack.push(0);
minStack.push(-3);
minStack.getMin(); --> 返回 -3.
minStack.pop();
minStack.top(); --> 返回 0.
minStack.getMin(); --> 返回 -2.

提示：

  • pop、top 和 getMin 操作总是在 非空栈 上调用。
class MinStack:
def __init__(self):
"""
initialize your data structure here.
"""
self.stack = []
self.miniStack = []
def push(self, x):
"""
:type x: int
:rtype: void
"""
if len(self.miniStack) == 0 or x <= self.miniStack[-1]:
self.miniStack.append(x)
self.stack.append(x)
def pop(self):
"""
:rtype: void
"""
if self.top() == self.getMin():
self.miniStack.pop()
self.stack.pop()
def top(self):
"""
:rtype: int
"""
return self.stack[-1]
def getMin(self):
"""
:rtype: int
"""
if len(self.miniStack) > 0:
return self.miniStack[-1]
return 0

# obj = MinStack()
# obj.push(x)
# obj.pop()
# param_3 = obj.top()
# param_4 = obj.getMin()

144. 二叉树的前序遍历#

非递归写法

# Definition for a binary tree node.
# class TreeNode:
# def __init__(self, x):
# self.val = x
# self.left = None
# self.right = None
class Solution:
def preorderTraversal(self, root: TreeNode) -> List[int]:
"""
使用栈数据结构,后进先出。
先序遍历（根节点，左子树，右子树）
"""
#先处理异常
if not root:
return []
stack, res = [root],[]
while stack:
node = stack.pop() # Python数组的pop操作默认移除尾部数据
if node:
res.append(node.val)
if node.right:
stack.append(node.right) # Python数组的append操作默认将数据添加到尾部
if node.left:
stack.append(node.left)
return res

非递归写法

# Definition for a binary tree node.
# class TreeNode:
# def __init__(self, x):
# self.val = x
# self.left = None
# self.right = None
class Solution:
def preorderTraversal(self, root: TreeNode) -> List[int]:
"""
根节点和左子树先入栈
"""
if not root:
return []
cur, stack, res = root, [], []
while cur or stack:
while cur:
res.append(cur.val) #节点值
stack.append(cur) #根节点入栈
cur = cur.left #向左子节点滑动
#指针滑动到左节点的为None的位置，出栈网上走，找右节点
tmp = stack.pop()
cur = tmp.right #找右节点
return res

class MyStack {
private:
queue<int> que_in;
queue<int> que_out;
public:
MyStack() {
}
void push(int x) {
//这个操作让新来的总是在前面。
que_in.push(x);
while(!que_out.empty()) {
que_in.push(que_out.front());
que_out.pop();
}
swap(que_in,que_out);//保证in总是空的
}
int pop() {
int ret = top();
if(!que_out.empty()) {
que_out.pop();
}
return ret;
}
int top() {
if(que_out.empty()) {
return -1;
}
int ret = que_out.front();
return ret;
}
bool empty() {
return que_in.empty() && que_out.empty();
}
};

103. 二叉树的锯齿形层序遍历#

/**
* Definition for a binary tree node.
* struct TreeNode {
* int val;
* TreeNode *left;
* TreeNode *right;
* TreeNode() : val(0), left(nullptr), right(nullptr) {}
* TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
* TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
* };
*/
class Solution {
public:
vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
vector<vector<int>> ret;
if(root == nullptr) {
return ret;
}
stack<TreeNode *> sk;//后进先出的辅助结构
sk.push(root);
bool to_left = true;
while(!sk.empty()) {
//针对每一层，有一个临时的数组和栈。
vector<int> levelRet;
stack<TreeNode *> levelSk;
while(!sk.empty()) {
TreeNode *node = sk.top();
sk.pop();
levelRet.push_back(node->val);
if(to_left){
if(node->left) levelSk.push(node->left);
if(node->right) levelSk.push(node->right);
}
else
{
if(node->right) levelSk.push(node->right);
if(node->left) levelSk.push(node->left);
}
}Reveal
to_left = !to_left;
sk = levelSk;
ret.push_back(levelRet);
}
return ret;
}
};
