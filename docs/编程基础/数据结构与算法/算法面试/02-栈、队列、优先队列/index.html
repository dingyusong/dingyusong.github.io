class Solution:
def isValid(self, s):
"""
:type s: str
:rtype: bool
"""
n = len(s)
listStack = []
for i in range(n):
if len(listStack) == 0:
listStack.append(s[i])
elif s[i] == ")" and listStack[-1] == "(":
listStack.pop()
elif s[i] == "}" and listStack[-1] == "{":
listStack.pop()
elif s[i] == "]" and listStack[-1] == "[":
listStack.pop()
else:
listStack.append(s[i])
if len(listStack) == 0:
return True
else:
return False
