auto key_value = *m_map[key];//拿到一个pair
m_cache.erase(m_map[key]);
m_cache.push_front(key_value);
m_map[key] = m_cache.begin();
return key_value.second;
}

class LRUCache {
private:
//注意pair的这个key同时也是hash的key
list<pair<int ,int >> m_cache;
unordered_map<int ,list<pair<int,int>>::iterator> m_hash;
int m_capacity;
public:
LRUCache(int capacity):m_capacity(capacity) {
}
int get(int key) {
auto it = m_hash.find(key);
//1. 没找到
if(it == m_hash.end()) return -1;
//2. 更新cache
m_cache.erase(it->second);
m_cache.push_front(*(it->second));
//3. 更新hash
m_hash[key] = m_cache.begin();
//4. 返回值
return it->second->second;
}
void put(int key, int value) {
auto it = m_hash.find(key);
//没找到
if(it == m_hash.end()) {
//那就需要添加了(添加的时候需要判断容量是否已满，如果满了就需要删除)
if(m_cache.size() == m_capacity) {
//删除hash
m_hash.erase(m_cache.back().first);
//删除cache
m_cache.pop_back();
}
}
else {
m_cache.erase(it->second);
}
//更新cache
m_cache.push_front({key,value});
//更新hash
m_hash[key] = m_cache.begin();
}
};

* int param_1 = obj->get(key);
* obj->put(key,value);
*/
class LRUCache:
def __init__(self, capacity):
"""
:type capacity: int
"""
self.cache = {}#保存key-value值
self.keys = []#保持key的位置
self.capacity = capacity
def get(self, key):
"""
:type key: int
:rtype: int
"""
#如果存在
if key in self.cache:
if key != self.keys[-1]:
#将key移到栈顶
self.keys.remove(key)
self.keys.append(key)
#返回value值
return self.cache[key]
else:#如果不存在返回-1
return -1
def put(self, key, value):
"""
:type key: int
:type value: int
:rtype: void
"""
if key in self.cache:#存在
#移除key
self.keys.remove(key)
elif len(self.keys) == self.capacity:#不存在且已经满了
#移除最近最远未使用
old = self.keys.pop(0)
self.cache.pop(old)
#1.更新位置
self.keys.append(key)
#2.更新值
self.cache[key] = value

1. 两数之和#

给定一个整数数组 nums 和一个目标值 target，请你在该数组中找出和为目标值的那 两个 整数，并返回他们的数组下标。

你可以假设每种输入只会对应一个答案。但是，数组中同一个元素不能使用两遍。

示例:

给定 nums = [2, 7, 11, 15], target = 9

因为 nums[0] + nums[1] = 2 + 7 = 9 所以返回 [0, 1]

class Solution {
public:
vector<int> twoSum(vector<int>& nums, int target) {
for(int i=0; i<nums.size(); i++) {
int numberA = nums[i];
for (int j=i+1; j<nums.size(); j++){
int numberB = nums[j];
int total = numberA+numberB;
if(target == total){
vector<int> result;
result.push_back(i);
result.push_back(j);
return result;
}
}
}
}
};
复杂度分析：
时间复杂度：O(n^2)。
对于每个元素，我们试图通过遍历数组的其余部分来寻找它所对应的目标元素，这将耗费 O(n)O(n) 的时间。因此时间复杂度为 O(n^2)。
空间复杂度：O(1)。

方法二：两遍哈希表 为了对运行时间复杂度进行优化，我们需要一种更有效的方法来检查数组中是否存在目标元素。如果存在，我们需要找出它的索引。保持数组中的每个元素与其索引相互对应的最好方法是什么？哈希表。

通过以空间换取速度的方式，我们可以将查找时间从 O(n)O(n) 降低到 O(1)O(1)。哈希表正是为此目的而构建的，它支持以 近似 恒定的时间进行快速查找。我用“近似”来描述，是因为一旦出现冲突，查找用时可能会退化到 O(n)O(n)。但只要你仔细地挑选哈希函数，在哈希表中进行查找的用时应当被摊销为 O(1)O(1)。

一个简单的实现使用了两次迭代。在第一次迭代中，我们将每个元素的值和它的索引添加到表中。然后，在第二次迭代中，我们将检查每个元素所对应的目标元素（target - nums[i]target−nums[i]）是否存在于表中。注意，该目标元素不能是 nums[i]nums[i] 本身！

class Solution {
public int[] twoSum(int[] nums, int target) {
Map<Integer, Integer> map = new HashMap<>();
for (int i = 0; i < nums.length; i++) {
map.put(nums[i], i);
}
for (int i = 0; i < nums.length; i++) {
int complement = target - nums[i];
if (map.containsKey(complement) && map.get(complement) != i) {
return new int[] { i, map.get(complement) };
}
}
throw new IllegalArgumentException("No two sum solution");
}
}
class Solution:
def twoSum(self, nums, target):
"""
:type nums: List[int]
:type target: int
:rtype: List[int]
"""
#hash表遍历
dict = {}
for i in range(len(nums)):
complement = target - nums[i]
if complement in dict:
return [dict[complement],i]
dict[nums[i]] = i

169. 多数元素#

class Solution:
def majorityElement(self, nums):
"""
:type nums: List[int]
:rtype: int
"""
#既然超过一半，那计数器肯定是不会归零的，其他的数加起来也消除不过他，最多消掉剩余1个
count = 0
maxNum = nums[0]
for i in range(len(nums)):
if maxNum == nums[i]:
count += 1
else:
count -= 1
if count == 0:
maxNum = nums[i+1]
return maxNum
