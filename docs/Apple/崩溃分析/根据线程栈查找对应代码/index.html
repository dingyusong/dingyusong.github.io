}

Ret = ErrorCode_FFmpeg_FrameInitError;
}

if ( Ret != ErrorCode_NoError){
return Ret;
}
}

Ret = this->InitFFmpegVideoConvertor(config);
if ( Ret != ErrorCode_NoError){
return Ret;
}
}
