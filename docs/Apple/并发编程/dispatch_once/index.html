dispatch_once

//dispatch_once_t 就是long
typedef intptr_t dispatch_once_t;
void
dispatch_once(dispatch_once_t *predicate,
DISPATCH_NOESCAPE dispatch_block_t block);
#define dispatch_once _dispatch_once
//实际调用
void
_dispatch_once(dispatch_once_t *predicate,
DISPATCH_NOESCAPE dispatch_block_t block)
{
//DISPATCH_EXPECT(*predicate, ~0l) 就是 *predicate 第一次进入就是0l
if (DISPATCH_EXPECT(*predicate, ~0l) != ~0l) {
dispatch_once(predicate, block);
} else {
dispatch_compiler_barrier();
}
DISPATCH_COMPILER_CAN_ASSUME(*predicate == ~0l);
}
#if __GNUC__
#define DISPATCH_EXPECT(x, v) __builtin_expect((x), (v))
#define dispatch_compiler_barrier() __asm__ __volatile__("" ::: "memory")
#else
#define DISPATCH_EXPECT(x, v) (x)
#define dispatch_compiler_barrier() do { } while (0)
#endif
/***********/
void
dispatch_once(dispatch_once_t *val, dispatch_block_t block)
{
/**
_dispatch_Block_invoke(block)
等价于：
((struct Block_layout *)bb) = block;
((dispatch_function_t)bb->invoke)
dispatch_function_t 就是 void * 就是一个函数指针！
*/
dispatch_once_f(val, block, _dispatch_Block_invoke(block));
}
#define _dispatch_Block_invoke(bb) \
((dispatch_function_t)((struct Block_layout *)bb)->invoke)
struct Block_layout {
void *isa;
volatile int32_t flags; // contains ref count
int32_t reserved;
void (*invoke)(void *, ...);
struct Block_descriptor_1 *descriptor;
// imported variables
};
typedef void (*dispatch_function_t)(void *_Nullable);
